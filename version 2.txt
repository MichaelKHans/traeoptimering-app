        let availableDimensions = ['45x95', '45x120', '45x145'];
        let requiredPieces = [];
        let stockLengths = [];
        let nextPieceId = 0;
        let nextStockId = 0;
        const STORAGE_KEY = 'traeOptAppProjects';

         --- DOM ELEMENT REFERENCES ---
        const projectNameInput = document.getElementById('project-name');
        const saveProjectBtn = document.getElementById('save-project-btn');
        const savedProjectsSelect = document.getElementById('saved-projects-select');
        const loadProjectBtn = document.getElementById('load-project-btn');
        const deleteProjectBtn = document.getElementById('delete-project-btn');
        
        const addDimensionForm = document.getElementById('add-dimension-form');
        const dimensionsListContainer = document.getElementById('dimensions-list');
        const pieceDimensionSelect = document.getElementById('piece-dimension-select');
        const stockDimensionSelect = document.getElementById('stock-dimension-select');
        const stockLengthSelect = document.getElementById('stock-length-select');
        
        const addPieceForm = document.getElementById('add-piece-form');
        const piecesListContainer = document.getElementById('pieces-list');
        const addStockForm = document.getElementById('add-stock-form');
        const stockListContainer = document.getElementById('stock-list');
        
        const calculateBtnDesktop = document.getElementById('calculate-btn-desktop');
        const printBtnDesktop = document.getElementById('print-btn-desktop');
        const resultsContainer = document.getElementById('results-container');
        const sustainabilityReportContainer = document.getElementById('sustainability-report');
        const safetyMarginInput = document.getElementById('safety-margin');

         --- COLOR PALETTE (Lighter Colors) ---
        const pieceColors = ['#a5b4fc', '#f9a8d4', '#86efac', '#fcd34d', '#7dd3fc', '#c4b5fd', '#fca5a5', '#6ee7b7', '#fdba74', '#67e8f9'];

        saveProjectBtn.addEventListener('click', handleSaveProject);

        deleteProjectBtn.addEventListener('click', handleDeleteProject);
        
        addDimensionForm.addEventListener('submit', handleAddDimension);

         --- PROJECT MANAGEMENT FUNCTIONS ---
        function getSavedProjects() {
            return JSON.parse(localStorage.getItem(STORAGE_KEY)  '{}');
        }

        function populateProjectsDropdown() {
            const projects = getSavedProjects();
            const projectNames = Object.keys(projects);
            if (projectNames.length === 0) {
                savedProjectsSelect.innerHTML = `optionIngen gemte projekteroption`;
                savedProjectsSelect.disabled = true;
                loadProjectBtn.disabled = true;
                deleteProjectBtn.disabled = true;
            } else {
                savedProjectsSelect.innerHTML = projectNames.map(name = `option value=${name}${name}option`).join('');
                savedProjectsSelect.disabled = false;
                loadProjectBtn.disabled = false;
                deleteProjectBtn.disabled = false;
            }
        }

        function handleSaveProject() {
            const name = projectNameInput.value.trim();
            if (!name) {
                alert('Indtast venligst et navn til dit projekt.');
                return;
            }
            const projects = getSavedProjects();
            projects[name] = {
                availableDimensions,
                requiredPieces,
                stockLengths,
                safetyMargin safetyMarginInput.value,
                nextPieceId,
                nextStockId
            };
            localStorage.setItem(STORAGE_KEY, JSON.stringify(projects));
            populateProjectsDropdown();
            alert(`Projektet '${name}' er blevet gemt.`);
        }

        function handleLoadProject() {
            const name = savedProjectsSelect.value;
            if (!name) return;
            const projects = getSavedProjects();
            const projectData = projects[name];
            if (projectData) {
                availableDimensions = projectData.availableDimensions  [];
                requiredPieces = projectData.requiredPieces  [];
                stockLengths = projectData.stockLengths  [];
                safetyMarginInput.value = projectData.safetyMargin  100;
                nextPieceId = projectData.nextPieceId  0;
                nextStockId = projectData.nextStockId  0;
                
                projectNameInput.value = name;
                renderAllLists();
                resultsContainer.innerHTML = '';  Clear results on load
                sustainabilityReportContainer.classList.add('hidden');
                showPrintButtons(false);
                alert(`Projektet '${name}' er blevet indlæst.`);
            }
        }

        function handleDeleteProject() {
            const name = savedProjectsSelect.value;
            if (!name) return;
            if (confirm(`Er du sikker på, at du vil slette projektet '${name}' Denne handling kan ikke fortrydes.`)) {
                const projects = getSavedProjects();
                delete projects[name];
                localStorage.setItem(STORAGE_KEY, JSON.stringify(projects));
                populateProjectsDropdown();
                alert(`Projektet '${name}' er blevet slettet.`);
            }
        }

         --- CORE APP FUNCTIONS ---
        function handleAddDimension(e) { e.preventDefault(); const name = new FormData(e.target).get('name').trim(); if (name && !availableDimensions.includes(name)) { availableDimensions.push(name); availableDimensions.sort(); renderAllLists(); } e.target.reset(); }
        function removeDimension(name) { availableDimensions = availableDimensions.filter(d = d !== name); requiredPieces = requiredPieces.filter(p = p.dimension !== name); stockLengths = stockLengths.filter(s = s.dimension !== name); renderAllLists(); }
        
        function handleAddPiece(e) {
            e.preventDefault();
            const fd = new FormData(e.target);
            const piece = {
                id nextPieceId++,
                dimension fd.get('dimension'),
                tag fd.get('tag').trim(),
                length parseInt(fd.get('length')),
                amount parseInt(fd.get('amount'))
            };
            if (!piece.dimension) { alert('Opretvælg en dimension først.'); return; }
            if (isNaN(piece.length)  isNaN(piece.amount)  piece.length = 0  piece.amount = 0) { alert('Ugyldig længdeantal.'); return; }
            requiredPieces.push(piece);
            renderPiecesList();
            e.target.reset();
            document.getElementById('piece-length').focus();
        }

        function handleAddStock(e) { e.preventDefault(); const fd = new FormData(e.target); const stock = { id nextStockId++, dimension fd.get('dimension'), length parseInt(fd.get('length')), amount parseInt(fd.get('amount')) }; if (!stock.dimension) { alert('Opretvælg en dimension først.'); return; } if (isNaN(stock.length)  isNaN(stock.amount)  stock.length = 0  stock.amount = 0) { alert('Ugyldig længdeantal.'); return; } stockLengths.push(stock); renderStockList(); e.target.reset(); document.getElementById('stock-amount').focus(); }
        function removePiece(id) { requiredPieces = requiredPieces.filter(p = p.id !== id); renderPiecesList(); }
        function removeStock(id) { stockLengths = stockLengths.filter(s = s.id !== id); renderStockList(); }
        
        function handleCalculate() {
            resultsContainer.innerHTML = '';
            sustainabilityReportContainer.classList.add('hidden');
            showPrintButtons(false);
            if (requiredPieces.length === 0) { renderError('Du skal tilføje mindst ét emne.'); return; }
            
            const safetyMargin = parseInt(safetyMarginInput.value)  0;
            try {
                 Run optimization with current stock
                const result = runOptimization(safetyMargin, stockLengths, requiredPieces);
                renderResults(result);

                 Run sustainability and purchase analysis
                const analysis = runAdvancedAnalysis(safetyMargin);
                
                renderSustainabilityReport(analysis, result.totalWaste);

                showPrintButtons(true);
            } catch (error) {
                renderError(error.message);
            }
        }

         --- RENDERING & UI FUNCTIONS ---
        function showPrintButtons(show) { if (show) { printBtnDesktop.classList.remove('hidden'); } else { printBtnDesktop.classList.add('hidden'); } }
        function renderAllLists() { renderDimensionsList(); updateDimensionDropdowns(); renderPiecesList(); renderStockList(); }
        function populateStandardLengthsDropdown() { const standardLengths = []; for (let l = 2400; l = 5400; l += 300) { standardLengths.push(l); } stockLengthSelect.innerHTML = standardLengths.map(len = `option value=${len}${len} mmoption`).join(''); }

        function updateDimensionDropdowns() { const optionsHTML = availableDimensions.map(dim = `option value=${dim}${dim}option`).join(''); pieceDimensionSelect.innerHTML = optionsHTML; stockDimensionSelect.innerHTML = optionsHTML; }
        
        function renderPiecesList() {
            piecesListContainer.innerHTML = '';
            if (requiredPieces.length === 0) {
                piecesListContainer.innerHTML = `p class=text-sm text-gray-500Ingen emner tilføjet.p`;
                return;
            }
            requiredPieces.forEach(p = {
                const item = document.createElement('div');
                item.className = 'flex justify-between items-center p-2 bg-gray-50 rounded-md border border-gray-200';
                item.innerHTML = `
                    div class=flex-grow
                        span class=font-mono text-sm text-gray-800${p.amount} stk. &nbsp;&nbsp; ${p.dimension} &nbsp;&nbsp; ${p.length} mmspan
                        p class=text-xs text-indigo-600 font-semibold truncate${p.tag  'Intet tag'}p
                    div
                    button onclick=removePiece(${p.id}) class=text-red-500 hovertext-red-700 font-semibold text-sm ml-4 flex-shrink-0Fjernbutton
                `;
                piecesListContainer.appendChild(item);
            });
        }

        function renderStockList() { stockListContainer.innerHTML = ''; if (stockLengths.length === 0) { stockListContainer.innerHTML = `p class=text-sm text-gray-500Ingen lagerlængder tilføjet.p`; return; } stockLengths.forEach(s = { const item = document.createElement('div'); item.className = 'flex justify-between items-center p-2 bg-gray-50 rounded-md border border-gray-200'; item.innerHTML = `span class=font-mono text-sm text-gray-800${s.amount} stk. &nbsp;&nbsp; ${s.dimension} &nbsp;&nbsp; ${s.length} mmspanbutton onclick=removeStock(${s.id}) class=text-red-500 hovertext-red-700 font-semibold text-smFjernbutton`; stockListContainer.appendChild(item); }); }

        function runOptimization(safetyMargin, currentStock, piecesToProcess) {
            const piecesSource = piecesToProcess;
            const piecesByDimension = {};
            piecesSource.forEach(p = {
                if (!piecesByDimension[p.dimension]) piecesByDimension[p.dimension] = [];
                const amount = p.amount  1;
                for (let i = 0; i  amount; i++) {
                    piecesByDimension[p.dimension].push({
                        originalLength p.length,
                        length p.length + safetyMargin,
                        margin safetyMargin,
                        tag p.tag
                    });
                }
            });

            const stockByDimension = {}; 
            (currentStock  []).forEach(s = { 
                if (!stockByDimension[s.dimension]) stockByDimension[s.dimension] = []; 
                for (let i = 0; i  s.amount; i++) { stockByDimension[s.dimension].push(s.length); } 
            }); 
            
            const finalPlan = {}; 
            let totalUsedStockLength = 0, totalRequiredLength = 0; 
            
            for (const dim in piecesByDimension) { 
                const pieces = piecesByDimension[dim]; 
                pieces.sort((a, b) = b.length - a.length); 
                const availableStockForDim = stockByDimension[dim]  [...stockByDimension[dim]]  []; 
                if (currentStock && (!availableStockForDim  availableStockForDim.length === 0)) { 
                    throw new Error(`Ingen lagerlængder for dimension ${dim}.`); 
                } 
                const usedStockForDim = []; 
                for (const piece of pieces) { 
                    totalRequiredLength += piece.length; 
                    let placed = false; 
                    usedStockForDim.sort((a,b) = a.remaining - b.remaining); 
                    for (const stock of usedStockForDim) { 
                        if (stock.remaining = piece.length) { 
                            stock.remaining -= piece.length; 
                            stock.pieces.push(piece); 
                            placed = true; 
                            break; 
                        } 
                    } 
                    if (!placed) { 
                        availableStockForDim.sort((a, b) = a - b); 
                        const bestFitIndex = availableStockForDim.findIndex(len = len = piece.length); 
                        
                        if (bestFitIndex === -1) { 
                            if (currentStock) {
                                throw new Error(`Ikke nok træ til emne på ${piece.length}mm for dimension ${dim}.`); 
                            } else {
                                const standardLengths = [];
                                for (let l = 2400; l = 5400; l += 300) { standardLengths.push(l); }
                                const newStockLength = standardLengths.find(l = l = piece.length);
                                if (!newStockLength) throw new Error(`Emne på ${piece.length}mm er for langt.`);
                                usedStockForDim.push({ originalLength newStockLength, remaining newStockLength - piece.length, pieces [piece] });
                            }
                        } else {
                            const newStockLength = availableStockForDim.splice(bestFitIndex, 1)[0]; 
                            usedStockForDim.push({ originalLength newStockLength, remaining newStockLength - piece.length, pieces [piece] }); 
                        }
                    } 
                } 
                finalPlan[dim] = usedStockForDim.map(stock = { 
                    totalUsedStockLength += stock.originalLength; 
                    return { originalLength stock.originalLength, pieces stock.pieces, waste stock.remaining }; 
                }); 
            } 
            const totalWaste = totalUsedStockLength - totalRequiredLength; 
            const wastePercentage = totalUsedStockLength  0  (totalWaste  totalUsedStockLength)  100  0; 
            return { plan finalPlan, totalUsedStockLength, totalRequiredLength, totalWaste, wastePercentage }; 
        }

        function runAdvancedAnalysis(safetyMargin) {
            const naiveWasteDetails = {};
            let totalNaiveWasteLength = 0;

            const allPiecesByDim = {};
            requiredPieces.forEach(p = {
                if (!allPiecesByDim[p.dimension]) {
                    allPiecesByDim[p.dimension] = [];
                }
                allPiecesByDim[p.dimension].push(p);
            });

            for (const dim in allPiecesByDim) {
                let naiveWasteForDim = 0;
                const piecesInDim = allPiecesByDim[dim];
                piecesInDim.forEach(p = {
                    const pieceLengthWithMargin = p.length + safetyMargin;
                    const standardLengths = [];
                    for (let l = 2400; l = 5400; l += 300) { standardLengths.push(l); }
                    const suitableStock = standardLengths.find(l = l = pieceLengthWithMargin);
                    if (suitableStock) {
                        naiveWasteForDim += (suitableStock - pieceLengthWithMargin)  p.amount;
                    }
                });
                naiveWasteDetails[dim] = naiveWasteForDim;
                totalNaiveWasteLength += naiveWasteForDim;
            }

            let purchaseSuggestion = {};
            for(const dim in allPiecesByDim) {
                let bestForDim = { waste Infinity, length 0, count 0 };
                const piecesForThisDim = allPiecesByDim[dim];
                const standardLengths = [];
                for (let l = 2400; l = 5400; l += 300) { standardLengths.push(l); }

                for(const stockLen of standardLengths) {
                    const tempStock = [];
                    for(let i = 0; i  500; i++) tempStock.push({dimension dim, length stockLen, amount 1}); 
                    
                    try {
                        const simResult = runOptimization(safetyMargin, tempStock, piecesForThisDim);
                        if(simResult.plan[dim] && simResult.totalWaste  bestForDim.waste) {
                            bestForDim.waste = simResult.totalWaste;
                            bestForDim.length = stockLen;
                            bestForDim.count = simResult.plan[dim].length;
                        }
                    } catch (e) {  ignore errors  }
                }
                purchaseSuggestion[dim] = bestForDim;
            }
            
            return { totalNaiveWasteLength, naiveWasteDetails, purchaseSuggestion };
        }
